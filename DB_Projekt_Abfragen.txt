1) 
Alle Mitarbeiter die einen Lieferauftrag betreuen, der noch nicht abgewickelt ist.


SELECT mitarbeiter.Name, mitarbeiter.Mitarbeiter_ID, mitarbeiter.Vorname, lieferantenbestellung.LAuftrag_ID 

FROM mitarbeiter JOIN lieferant ON mitarbeiter.Mitarbeiter_ID = lieferant.Mitarbeiter_ID

JOIN lieferantenbestellung ON lieferant.Lieferant_ID = lieferantenbestellung.Lieferant_ID

WHERE lieferantenbestellung.Status = 0;

Alle Mitarbeiter die einen Lieferauftrag betreut haben.

SELECT mitarbeiter.Name, mitarbeiter.Mitarbeiter_ID, mitarbeiter.Vorname, lieferantenbestellung.LAuftrag_ID 

FROM mitarbeiter JOIN lieferant ON mitarbeiter.Mitarbeiter_ID = lieferant.Mitarbeiter_ID

JOIN lieferantenbestellung ON lieferant.Lieferant_ID = lieferantenbestellung.Lieferant_ID

WHERE lieferantenbestellung.Status = 1;


2)
Alle Mitarbeiter die eine Bestellung betreuen, die noch nicht abgewickelt ist.


SELECT mitarbeiter.Mitarbeiter_ID, mitarbeiter.Name, mitarbeiter.Vorname, kunde.Kunde_ID, kunde.Name, kundenbestellung.Auftrag_ID

FROM mitarbeiter JOIN kundenbestellung ON mitarbeiter.Mitarbeiter_ID = kundenbestellung.Mitarbeiter_ID

JOIN Kunde ON kundenbestellung.Kunde_ID = kunde.Kunde_ID

WHERE kundenbestellung.Status = 0;

Alle Mitarbeiter die eine Bestellung erfolgreich betreut haben

SELECT mitarbeiter.Mitarbeiter_ID, mitarbeiter.Name, mitarbeiter.Vorname, kunde.Kunde_ID, kunde.Name, kundenbestellung.Auftrag_ID

FROM mitarbeiter JOIN kundenbestellung ON mitarbeiter.Mitarbeiter_ID = kundenbestellung.Mitarbeiter_ID

JOIN Kunde ON kundenbestellung.Kunde_ID = kunde.Kunde_ID

WHERE kundenbestellung.Status = 1;


3)
Welches produkt muss nachbestellt werden?

SELECT produkt.Name, produktkategorie.Kategorie, produkt.Lagerbestand, produkt.Mindestbestand

FROM produkt JOIN produktkategorie ON produkt.Kategorie_ID = produktkategorie.Kategorie_ID

WHERE produkt.Lagerbestand < produkt.Mindestbestand;


4)
Alle Kundenbestellungen

SELECT kunde.Name, kunde.Vorname, kundenbestellung.Auftrag_ID, artikel.Name

FROM kunde JOIN kundenbestellung ON kunde.Kunde_ID = kundenbestellung.Kunde_ID

JOIN artikel ON kundenbestellung.Auftrag_ID = artikel.Auftrag_ID

ORDER BY kundenbestellung.Auftrag_ID ASC;


5)
Kunden die etwas aus der Kategorie Motor bestellt haben

SELECT kunde.Kunde_ID, kunde.Name, kunde.Vorname FROM kunde WHERE kunde.Kunde_ID IN 
	
	(SELECT kundenbestellung.Kunde_ID FROM kundenbestellung WHERE kundenbestellung.Auftrag_ID IN 
		
		(SELECT artikel.Auftrag_ID FROM artikel WHERE artikel.Artikel_ID IN 
			
			(SELECT produkt_artikel.Artikel_ID FROM produkt_artikel WHERE produkt_artikel.Produkt_ID IN 
				
				(SELECT produkt.Produkt_ID FROM produkt WHERE produkt.Kategorie_ID IN 
					
					(SELECT produktkategorie.Kategorie_ID FROM produktkategorie WHERE produktkategorie.Kategorie LIKE "%Motor%")))))


6)
Welcher Kunde wird von wem beliefert?


Select kunde.Kunde_id,kunde.Name, lieferant.Name From kunde 
Join kundenbestellung 
ON kunde.Kunde_ID = kundenbestellung.Kunde_ID 
Join artikel
 ON kundenbestellung.Auftrag_ID = artikel.Auftrag_ID 
JOIN produkt_artikel 
ON artikel.Artikel_ID = produkt_artikel.Artikel_ID 
JOIN produkt 
ON produkt_artikel.Produkt_ID = produkt.Produkt_ID 
JOIN lieferant
 ON produkt.Lieferant_ID = lieferant.Lieferant_ID
